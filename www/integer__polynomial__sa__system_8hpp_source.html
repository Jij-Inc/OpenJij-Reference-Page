30 IntegerSASystem(const graph::IntegerPolynomialModel &model,
31 const typename RandType::result_type seed)
32 : model(model), seed(seed), random_number_engine(seed) {
33
34 const std::int64_t num_variables = model.GetNumVariables();
35
36 // Initialize the state with random values
37 this->state_.reserve(num_variables);
38 const auto &bounds = this->model.GetBounds();
39 for (std::int64_t i = 0; i < num_variables; ++i) {
40 auto x = utility::IntegerVariable(bounds[i].first, bounds[i].second);
41 x.SetRandomValue(this->random_number_engine);
42 this->state_.push_back(x);
43 }
44 this->energy_ = this->CalculateEnergy(this->state_);
45
46 // Set zero count and term product
47 const auto &key_value_list = this->model.GetKeyValueList();
48 const std::size_t num_terms = key_value_list.size();
49 this->zero_count_.resize(num_terms, 0);
50 this->term_prod_.resize(num_terms, 1.0);
51
52 for (std::size_t i = 0; i < num_terms; ++i) {
53 int count = 0;
54 double prod = 1.0;
55 for (const auto &[index, degree] : key_value_list[i].first) {
56 auto x = this->state_[index].value;
57 if (x == 0) {
58 count++;
59 } else {
60 prod *= std::pow(x, degree);
61 }
62 }
63 this->zero_count_[i] = count;
64 this->term_prod_[i] = prod;
65 }
66
67 // Initialize energy difference
68 this->base_energy_difference_.resize(num_variables);
69 for (std::size_t i = 0; i < num_terms; ++i) {
70 const double value = key_value_list[i].second;
71 for (const auto &[index, degree] : key_value_list[i].first) {
72 const auto x = this->state_[index].value;
73 if (x == 0) {
74 if (this->zero_count_[i] == 1) {
75 this->base_energy_difference_[index][degree] +=
76 value * this->term_prod_[i];
77 }
78 } else {
79 if (this->zero_count_[i] == 0) {
80 this->base_energy_difference_[index][degree] +=
81 value * this->term_prod_[i] / std::pow(x, degree);
82 }
83 }
84 }
85 }
86 }

88 double
89 CalculateEnergy(const std::vector<utility::IntegerVariable> &state) const {
90 double energy = this->model.GetConstant();
91 const auto &key_value_list = this->model.GetKeyValueList();
92 for (const auto &term : key_value_list) {
93 double prod = 1.0;
94 for (const auto &[index, degree] : term.first) {
95 prod *= std::pow(state[index].value, degree);
96 }
97 energy += term.second * prod;
98 }
99 return energy;
100 }

107 double GetEnergyDifference(std::int64_t index, std::int64_t new_value) const {
108 double dE = 0.0;
109 const auto current_value = this->state_[index].value;
110 for (const auto &[degree, value] : this->base_energy_difference_[index]) {
111 dE += value *
112 (std::pow(new_value, degree) - std::pow(current_value, degree));
113 }
114 return dE;
115 }

117 void SetValue(std::int64_t index, std::int64_t new_value) {
118 const auto current_value = this->state_[index].value;
119 if (current_value == new_value) {
120 return;
121 }
122
123 this->energy_ += this->GetEnergyDifference(index, new_value);
124 this->state_[index].SetValue(new_value);
125
126 const auto &interactions = this->model.GetIndexToInteractions().at(index);
127 const auto &key_value_list = this->model.GetKeyValueList();
128
129 if (current_value != 0 && new_value != 0) {
130 for (const auto &[cons_ind, degree] : interactions) {
131 const double a =
132 std::pow(static_cast<double>(new_value) / current_value, degree);
133 const double ddE = (a - 1) * key_value_list[cons_ind].second *
134 this->term_prod_[cons_ind];
135 this->term_prod_[cons_ind] *= a;
136 const int total_count = this->zero_count_[cons_ind];
137 for (const auto &[i, d] : key_value_list[cons_ind].first) {
138 if (i != index) {
139 if (this->state_[i].value == 0 && total_count == 1) {
140 this->base_energy_difference_[i][d] += ddE;
141 } else if (this->state_[i].value != 0 && total_count == 0) {
142 this->base_energy_difference_[i][d] +=
143 ddE / std::pow(this->state_[i].value, d);
144 }
145 }
146 }
147 }
148 } else if (current_value == 0 && new_value != 0) {
149 for (const auto &[cons_ind, degree] : interactions) {
150 this->zero_count_[cons_ind]--;
151 const int total_count = this->zero_count_[cons_ind];
152 const double b = std::pow(new_value, degree);
153 const double ddE =
154 b * key_value_list[cons_ind].second * this->term_prod_[cons_ind];
155 this->term_prod_[cons_ind] *= b;
156 for (const auto &[i, d] : key_value_list[cons_ind].first) {
157 if (i != index) {
158 if (this->state_[i].value == 0 && total_count == 1) {
159 this->base_energy_difference_[i][d] += ddE;
160 } else if (this->state_[i].value != 0 && total_count == 0) {
161 this->base_energy_difference_[i][d] +=
162 ddE / std::pow(this->state_[i].value, d);
163 }
164 }
165 }
166 }
167 } else { // current_value != 0 && new_value == 0
168 for (const auto &[cons_ind, degree] : interactions) {
169 const int total_count = this->zero_count_[cons_ind];
170 this->zero_count_[cons_ind]++;
171 const double ddE =
172 -key_value_list[cons_ind].second * this->term_prod_[cons_ind];
173 this->term_prod_[cons_ind] /= std::pow(current_value, degree);
174 for (const auto &[i, d] : key_value_list[cons_ind].first) {
175 if (i != index) {
176 if (this->state_[i].value == 0 && total_count == 1) {
177 this->base_energy_difference_[i][d] += ddE;
178 } else if (this->state_[i].value != 0 && total_count == 0) {
179 this->base_energy_difference_[i][d] +=
180 ddE / std::pow(this->state_[i].value, d);
181 }
182 }
183 }
184 }
185 }

188 std::pair<int, double> GetMinEnergyDifference(std::int64_t index) {
189 if (this->UnderQuadraticCoeff(index)) {
190 const auto &x = this->state_[index];
191 const std::int64_t dxl = x.lower_bound - x.value;
192 const std::int64_t dxu = x.upper_bound - x.value;
193
194 auto it_a = this->base_energy_difference_[index].find(2);
195 const double a = (it_a != this->base_energy_difference_[index].end())
196 ? it_a->second
197 : 0.0;
198
199 auto it_b = this->base_energy_difference_[index].find(1);
200 const double b_base = (it_b != this->base_energy_difference_[index].end())
201 ? it_b->second
202 : 0.0;
203 const double b = b_base + 2 * x.value * a;
204
205 if (a > 0) {
206 const double center = -b / (2 * a);
207 if (dxu <= center) {
208 return {x.upper_bound,
209 this->GetEnergyDifference(index, x.upper_bound)};
210 } else if (dxl < center && center < dxu) {
211 const std::int64_t dx_left =
212 static_cast<std::int64_t>(std::floor(center));
213 const std::int64_t dx_right =
214 static_cast<std::int64_t>(std::ceil(center));
215 if (center - dx_left <= dx_right - center) {
216 return {x.value + dx_left,
217 this->GetEnergyDifference(index, x.value + dx_left)};
218 } else {
219 return {x.value + dx_right,
220 this->GetEnergyDifference(index, x.value + dx_right)};
221 }
222 } else if (dxl >= center) {
223 return {x.lower_bound,
224 this->GetEnergyDifference(index, x.lower_bound)};
225 } else {
226 throw std::runtime_error("Invalid state in GetMinEnergyDifference");
227 }
228 } else if (a == 0) {
229 if (b > 0) {
230 return {x.lower_bound,
231 this->GetEnergyDifference(index, x.lower_bound)};
232 } else if (b < 0) {
233 return {x.upper_bound,
234 this->GetEnergyDifference(index, x.upper_bound)};
235 } else {
236 return {x.GenerateRandomValue(this->random_number_engine), 0.0};
237 }
238 } else { // a < 0
239 const double dE_lower = this->GetEnergyDifference(index, x.lower_bound);
240 const double dE_upper = this->GetEnergyDifference(index, x.upper_bound);
241 if (dE_lower <= dE_upper) {
242 return {x.lower_bound, dE_lower};
243 } else {
244 return {x.upper_bound, dE_upper};
245 }
246 }
247 } else {
248 double min_dE = std::numeric_limits<double>::infinity();
249 std::int64_t min_value = -1;
250
251 for (std::int64_t val = this->state_[index].lower_bound;
252 val <= this->state_[index].upper_bound; ++val) {
253 double dE = this->GetEnergyDifference(index, val);
254 if (dE < min_dE) {
255 min_dE = dE;
256 min_value = val;
257 }
258 }
259 if (min_value == -1) {
260 throw std::runtime_error("No valid state number found.");
261 }
262 return {min_value, min_dE};
263 }
264 }

266 double GetLinearCoeff(std::int64_t index) const {
267 auto it1 = this->base_energy_difference_[index].find(1);
268 double c1 =
269 (it1 != this->base_energy_difference_[index].end()) ? it1->second : 0.0;
270
271 auto it2 = this->base_energy_difference_[index].find(2);
272 double c2 =
273 (it2 != this->base_energy_difference_[index].end()) ? it2->second : 0.0;
274
275 return c1 + 2 * this->state_[index].value * c2;
276 }
