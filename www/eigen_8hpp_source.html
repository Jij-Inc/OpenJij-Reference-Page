164  T t_list;
165 
166  for(size_t ind=0; ind<graph.get_num_spins(); ind++){
167  for(size_t adj_ind : graph.adj_nodes(ind)){
168  if(ind != adj_ind){
169  t_list.emplace_back(ind, adj_ind, graph.J(ind, adj_ind));
170  }
171  else{
172  t_list.emplace_back(ind, graph.get_num_spins(), graph.h(ind));
173  t_list.emplace_back(graph.get_num_spins(), ind, graph.h(ind));
174  }

179

181 
182  return ret_mat;
