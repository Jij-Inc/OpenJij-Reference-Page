57  template <typename RandomNumberEngine>
58  static void update(system::ContinuousTimeIsing<GraphType>& system,
59  RandomNumberEngine& random_number_engine,
60  const utility::TransverseFieldUpdaterParameter& parameter) {
61 
62  const graph::Index num_spin = system.num_spins;
63  std::vector<graph::Index> index_helper;
64  index_helper.reserve(num_spin+1);
65  index_helper.push_back(0);
66  /* index_helper[i]+k gives 1 dimensionalized index of kth time point at ith site.
67  * this helps use of union-find tree only available for 1D structure.
68  */
69 
70  /* 1. remove old cuts and place new cuts for every site */
71  for(graph::Index i = 0;i < num_spin;i++) {
72  auto& timeline = system.spin_config[i];
73  auto cuts = generate_poisson_points(0.5*system.gamma*(1.0-parameter.s), parameter.beta, random_number_engine);
74  // assuming transverse field gamma is positive
75 
76  timeline = create_timeline(timeline, cuts);
77  assert(timeline.size() > 0);
78 
79  index_helper.push_back(index_helper.back()+timeline.size());
80  }
81 
82  /* 2. place spacial bonds */
83  utility::UnionFind union_find_tree(index_helper.back());
84  for(graph::Index i = 0;i < num_spin;i++) {
85  for(typename CTIsing::SparseMatrixXx::InnerIterator it(system.interaction, i); it; ++it) {
86  std::size_t j = it.index();
87  const FloatType& J = it.value();
88  if (i < j) {
89  continue; // ignore duplicated interaction
90  // if adj_nodes are sorted, this "continue" can be replaced by "break"
91  }
92 
93  const auto bonds = generate_poisson_points(std::abs(0.5*J*parameter.s),
94  parameter.beta, random_number_engine);
95  for(const auto bond : bonds) {
96  /* get time point indices just before the bond */
97  auto ki = system.get_temporal_spin_index(i, bond);
98  auto kj = system.get_temporal_spin_index(j, bond);
99 
100  if(system.spin_config[i][ki].second * system.spin_config[j][kj].second * J < 0) {
101  union_find_tree.unite_sets(index_helper[i]+ki, index_helper[j]+kj);
102  }
103  }
104  }
105  }
106 
107  /* 3. make clusters */
108  std::unordered_map<utility::UnionFind::Node, std::vector<utility::UnionFind::Node>> cluster_map;
109  // root-index to {contained nodes} map
110 
111  for(graph::Index i = 0;i < num_spin;i++) {
112  for(size_t k = 0;k < system.spin_config[i].size();k++) {
113  auto index = index_helper[i] + k;
114  auto root_index = union_find_tree.find_set(index);
115  auto position = cluster_map.find(root_index);
116  if(position == cluster_map.end()) {
117  cluster_map.emplace(root_index, std::vector<utility::UnionFind::Node>{ index });
118  } else {
119  position->second.push_back(index);
120  }
121  }
122  }
123 
124  /* 4. flip clusters */
125  auto urd = std::uniform_real_distribution<>(0, 1.0);
126  for(const auto& cluster : cluster_map) {
127  // 4.1. decide spin state (flip with the probability 1/2)
128  const FloatType probability = 1.0 / 2.0;
129  if(urd(random_number_engine) < probability){
130  // 4.2. update spin states
131  for(auto node : cluster.second) {
132  auto i = std::distance(index_helper.begin(),
133  std::upper_bound(index_helper.begin(), index_helper.end(), node)) - 1;
134  auto k = node - index_helper[i];
135  system.spin_config[i][k].second *= -1;
136  }
137  }
138  }
139  }

145  static std::vector<CutPoint> create_timeline(const std::vector<CutPoint>& old_timeline,
146  const std::vector<TimeType>& cuts) {
147  /* remove redundant cuts*/
148  std::vector<CutPoint> concatenated_timeline;
149  auto current_spin = old_timeline.back().second;
150  for(auto cut_point : old_timeline) {
151  if(cut_point.second != current_spin) {
152  concatenated_timeline.push_back(cut_point);
153  }
154  current_spin = cut_point.second;
155  }
156 
157  /* if entire timeline is occupied by single spin state */
158  std::vector<CutPoint> new_timeline;
159  if(concatenated_timeline.empty()) {
160  if(cuts.empty()) {
161  new_timeline.push_back(old_timeline[0]);
162  } else {
163  for(auto cut : cuts) {
164  new_timeline.push_back(CutPoint(cut, current_spin));
165  }
166  }
167  return new_timeline;
168  }
169 
170  current_spin = concatenated_timeline.back().second;
171  auto timeline_itr = concatenated_timeline.begin();
172  auto cuts_itr = cuts.begin();
173  while(true) {
174  /* if all cuts have been placed, add remaining old kinks and break loop */
175  if(cuts_itr == cuts.end()) {
176  std::for_each(timeline_itr, concatenated_timeline.end(), [&](CutPoint it) {
177  new_timeline.push_back(it);
178  });
179  break;
180  }
181 
182  /* if all spin kinks have been placed, add remaining cuts and break loop */
183  if(timeline_itr == concatenated_timeline.end()) {
184  std::for_each(cuts_itr, cuts.end(), [&](TimeType it) {
185  new_timeline.push_back(CutPoint(it, current_spin));
186  });
187  break;
188  }
189 
190  /* add earlier of kink or cut to new timeline */
191  if(*cuts_itr < timeline_itr->first) {
192  new_timeline.push_back(CutPoint(*cuts_itr, current_spin));
193  cuts_itr++;
194  } else {
195  new_timeline.push_back(*timeline_itr);
196  current_spin = timeline_itr->second;
197  timeline_itr++;
198  }
199  }
200 
201  return new_timeline;
202  }

208  static std::vector<CutPoint> create_timeline_easy(const std::vector<CutPoint>& old_timeline,
209  const std::vector<TimeType>& cuts) {
210  /* remove redundant cuts*/
211  std::vector<CutPoint> new_timeline;
212  auto current_spin = old_timeline.back().second;
213  for(auto cut_point : old_timeline) {
214  if(cut_point.second != current_spin) {
215  new_timeline.push_back(cut_point);
216  }
217  current_spin = cut_point.second;
218  }
219 
220  /* if entire timeline is occupied by single spin state */
221  if(new_timeline.empty()) {
222  if(cuts.empty()) {
223  new_timeline.push_back(old_timeline[0]);
224  } else {
225  for(auto cut : cuts) {
226  new_timeline.push_back(CutPoint(cut, current_spin));
227  }
228  }
229  return new_timeline;
230  }
231 
232  static const auto first_lt = [](CutPoint x, CutPoint y) { return x.first < y.first; };
233  for(auto cut : cuts) {
234  const auto dummy_cut = CutPoint(cut, 0); // dummy cut for binary search
235  auto itr = std::upper_bound(new_timeline.begin(),
236  new_timeline.end(),
237  dummy_cut,
238  first_lt);
239  auto prev_itr = (itr == new_timeline.begin()) ? new_timeline.end() - 1 : itr - 1;
240 
241  new_timeline.insert(itr, CutPoint(cut, prev_itr->second));
242  }
243 
244  return new_timeline;
245  }

252  template<typename RandomNumberEngine>
253  static std::vector<TimeType> generate_poisson_points(const TimeType lambda, const TimeType beta,
254  RandomNumberEngine& random_number_engine) {
255  std::uniform_real_distribution<> rand(0.0, 1.0);
256  std::uniform_real_distribution<> rand_beta(0.0, beta);
257 
258  const TimeType coef = beta*lambda;
259  TimeType n = 0;
260  TimeType d = std::exp(-coef);
261  TimeType p = d;
262  TimeType xi = rand(random_number_engine);
263 
264  while(p < xi) {
265  n += 1;
266  d *= coef/n;
267  p += d;
268  }
269 
270  std::vector<TimeType> poisson_points(n);
271  for(int k = 0;k < n;k++) {
272  poisson_points[k] = rand_beta(random_number_engine);
273  }
274  std::sort(poisson_points.begin(), poisson_points.end());
275 
276  return poisson_points;
277  }
