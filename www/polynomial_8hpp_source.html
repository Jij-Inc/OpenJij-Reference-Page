57  explicit Polynomial(const nlohmann::json &j): Graph(j.at("variables").size()) {
58  const auto &v_k_v = json_parse_polynomial<FloatType>(j);
59  const auto &poly_key_list = std::get<0>(v_k_v);
60  const auto &poly_value_list = std::get<1>(v_k_v);
61 
62  if (poly_key_list.size() != poly_value_list.size()) {
63  throw std::runtime_error("The sizes of key_list and value_list must match each other");
64  }
65 
66  int64_t num_interactions = static_cast<int64_t>(poly_key_list.size());
67 
68  poly_key_list_.resize(num_interactions);
69  poly_value_list_.resize(num_interactions);
70 
71 #pragma omp parallel for
72  for (int64_t i = 0; i < num_interactions; ++i) {
73  poly_key_list_[i] = poly_key_list[i];
74  poly_value_list_[i] = poly_value_list[i];
75  }

77  for (std::size_t i = 0; i < poly_key_list.size(); ++i) {

182  FloatType energy(const Spins &spins, const bool omp_flag = true) const {
183  if(spins.size() != Graph::size()){
184  throw std::out_of_range("The size of spins/binaries does not equal to the size of polynomial graph");
185  }
186 
187  FloatType energy = 0.0;
188 
189  int64_t num_interactions = static_cast<int64_t>(poly_key_list_.size());
190 
191  if (omp_flag) {
192 #pragma omp parallel for reduction (+: energy)
193  for (int64_t i = 0; i < num_interactions; ++i) {
194  Spin spin_multiple = 1;
195  for (const auto &index: poly_key_list_[i]) {
196  spin_multiple *= spins[index];
197  if (spin_multiple == 0.0) {
198  break;
199  }
200  }
201  energy += spin_multiple*poly_value_list_[i];
202  }
203  }
204  else {
205  for (int64_t i = 0; i < num_interactions; ++i) {
206  Spin spin_multiple = 1;
207  for (const auto &index: poly_key_list_[i]) {
208  spin_multiple *= spins[index];
209  if (spin_multiple == 0.0) {
210  break;
211  }
212  }
213  energy += spin_multiple*poly_value_list_[i];
214  }
215  }
216  return energy;
217  }

239  void CheckKeyValid(const std::vector<Index> &key) const {
240  if (key.size() > Graph::size()) {
241  std::stringstream ss;
242  ss << "Too small system size. ";
243  ss << "The degree of the input polynomial interaction is " << key.size();
244  ss << ". But the system size is " << Graph::size();
245  throw std::runtime_error(ss.str());
246  }
247  if (0 < key.size()) {
248  //key is assumed to be sorted
249  for (std::size_t i = 0; i < key.size() - 1; ++i) {
250  if (key[i] == key[i + 1]) {
251  throw std::runtime_error("No self-loops allowed");
252  }
253  if (key[i] >= Graph::size()) {
254  std::stringstream ss;
255  ss << "Too small system size. ";
256  ss << "The index of a interaction: " << key[i] << " is out of range";
257  throw std::runtime_error(ss.str());
258  }
259  }
260  if (key.back() >= Graph::size()) {
261  std::stringstream ss;
262  ss << "Too small system size. ";
263  ss << "The index of a interaction: " << key.back() << " is out of range";
264  throw std::runtime_error(ss.str());
265  }
266  }
267  }
