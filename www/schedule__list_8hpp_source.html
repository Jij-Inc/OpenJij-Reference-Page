167  ClassicalScheduleList make_classical_schedule_list(double beta_min, double beta_max, std::size_t one_mc_step, std::size_t num_call_updater) {
168  double r_beta = std::pow(beta_max/beta_min, 1.0/static_cast<double>(num_call_updater - 1));
169  double beta = beta_min;
170 
171  auto schedule_list = ClassicalScheduleList(num_call_updater);
172  for (auto& schedule : schedule_list) {
173  schedule.one_mc_step = one_mc_step;
174  schedule.updater_parameter = ClassicalUpdaterParameter(beta);
175  beta *= r_beta;
176  }
177 
178  return schedule_list;
179  }

190  TransverseFieldScheduleList make_transverse_field_schedule_list(double beta, std::size_t one_mc_step, std::size_t num_call_updater) {
191  double ds = 1.0/static_cast<double>(num_call_updater - 1);
192  double s = 0;
193 
194  auto schedule_list = TransverseFieldScheduleList(num_call_updater);
195  for (auto& schedule : schedule_list) {
196  schedule.one_mc_step = one_mc_step;
197  schedule.updater_parameter = TransverseFieldUpdaterParameter(beta, s);
198  s += ds;
199  }

201  return schedule_list;
