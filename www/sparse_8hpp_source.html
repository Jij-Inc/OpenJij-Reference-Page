236  FloatType energy(const Spins& spins) const{
237  if(!(spins.size() == this->get_num_spins())){
238  std::out_of_range("Out of range in energy in Sparse graph.");
239  }
240 
241  FloatType ret = 0;
242  for(std::size_t ind=0; ind<this->get_num_spins(); ind++){
243  for(auto& adj_ind : _list_adj_nodes[ind]){
244  if(ind != adj_ind)
245  ret += (1./2) * this->J(ind, adj_ind) * spins[ind] * spins[adj_ind];
246  else
247  ret += this->h(ind) * spins[ind];
248  }
249  }
250 
251  return ret;
