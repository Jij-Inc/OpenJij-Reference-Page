65  template<typename RandomNumberEngine>
66  inline static void update(ClIsing& system,
67  RandomNumberEngine& random_number_engine,
68  const utility::ClassicalUpdaterParameter& parameter) {
69  // set probability distribution object
70  // to do Metroopolis
71  auto urd = std::uniform_real_distribution<>(0, 1.0);
72 
73  Eigen::setNbThreads(1);
74  Eigen::initParallel();
75 
76  // do a iteraction except for the auxiliary spin
77  for (std::size_t index = 0; index < system.num_spins; ++index) {
78 
79  if (system.dE(index) <= 0 || std::exp( -parameter.beta * system.dE(index)) > urd(random_number_engine)) {
80  // update dE
81  system.dE += 4 * system.spin(index) * (system.interaction.row(index).transpose().cwiseProduct(system.spin));
82 
83  system.dE(index) *= -1;
84  system.spin(index) *= -1;
85  }
86 
87  //assure that the dummy spin is not changed.
88  system.spin(system.num_spins) = 1;
89  }
90  }
91

121  template<typename RandomNumberEngine>
122  inline static void update(QIsing& system,
123  RandomNumberEngine& random_number_engine,
124  const utility::TransverseFieldUpdaterParameter& parameter) {
125 
126  //get number of classical spins
127  std::size_t num_classical_spins = system.num_classical_spins;
128  //get number of trotter slices
129  std::size_t num_trotter_slices = system.trotter_spins.cols();
130 
131  //do metropolis
132  auto urd = std::uniform_real_distribution<>(0, 1.0);
133 
134  //aliases
135  //const auto& spins = system.trotter_spins;
136  const auto& gamma = system.gamma;
137  const auto& beta = parameter.beta;
138  const auto& s = parameter.s;
139 
140  // transverse factor
141  FloatType B = (1/2.) * log(tanh(beta* gamma * (1.0-s) /num_trotter_slices));
142 
143  Eigen::setNbThreads(1);
144  Eigen::initParallel();
145 
146  //generate random number (col major)
147  for(std::size_t t=0; t<num_trotter_slices; t++){
148  for(std::size_t i=0; i<num_classical_spins; i++){
149  system.rand_pool(i, t) = urd(random_number_engine);
150  }
151  }
152 
153  //using OpenMP
154 
155  //we have to consider the case num_trotter_slices is odd.
156  std::size_t upper_limit = num_trotter_slices%2!=0 ? num_trotter_slices-1 : num_trotter_slices;
157 
158  //NOTE: only signed integer is allowed for OpenMP
159 #pragma omp parallel for
160  for(int64_t t=0; t<(int64_t)upper_limit; t+=2){
161  for(int64_t i=0; i<(int64_t)num_classical_spins; i++){
162  //calculate matrix dot product
163  do_calc(system, parameter, i, t, B);
164  }
165  }
166 
167  //using OpenMP
168 #pragma omp parallel for
169  for(int64_t t=1; t<(int64_t)num_trotter_slices; t+=2){
170  for(int64_t i=0; i<(int64_t)num_classical_spins; i++){
171  //calculate matrix dot product
172  do_calc(system, parameter, i, t, B);
173  }
174  }
175 
176  //for the case num_trotter_slices is odd.
177  if(num_trotter_slices%2!=0){
178  int64_t t = (int64_t)(num_trotter_slices-1);
179  for(int64_t i=0; i<(int64_t)num_classical_spins; i++){
180  //calculate matrix dot product
181  do_calc(system, parameter, i, t, B);
182  }
183  }
184  }

186 private:
187  inline static void do_calc(QIsing& system,
188  const utility::TransverseFieldUpdaterParameter& parameter, size_t i, size_t t, FloatType B) {
189 
190  //get number of trotter slices
191  std::size_t num_trotter_slices = system.trotter_spins.cols();
192 
193  //aliases
194  auto& spins = system.trotter_spins;
195  //const auto& gamma = system.gamma;
196  const auto& beta = parameter.beta;
197  const auto& s = parameter.s;
198 
199  //calculate dE for trotter direction
200  FloatType dEtrot = -2 * spins(i, t)*
201  (spins(i, mod_t((int64_t)t+1, num_trotter_slices)) +
202  spins(i, mod_t((int64_t)t-1, num_trotter_slices))
203  );
204 
205  //calculate total dE
206  FloatType dE = s * (beta/num_trotter_slices) * system.dE(i, t) + B * dEtrot;
207 
208  // for debugging
209 
210  //FloatType testdE = 0;
211 
213  //testdE += -2 * s * (beta/num_trotter_slices) * spins(i, t)*(system.interaction.row(i).dot(spins.col(t)));
214 
216  //testdE += -2 * (1/2.) * log(tanh(beta* gamma * (1.0-s) /num_trotter_slices)) * spins(i, t) *
217  // ( spins(i, mod_t((int64_t)t+1, num_trotter_slices))
218  // + spins(i, mod_t((int64_t)t-1, num_trotter_slices)));
219 
220  //std::cout << "s=" << s << std::endl;
221  //std::cout << "dE=" << dE << std::endl;
222  //std::cout << "testdE=" << testdE << std::endl;
223  //assert((std::isinf(dE) && std::isinf(testdE)) || abs(dE-testdE) < 1e-5);
224 
225  //metropolis
226 
227  if(dE < 0 || exp(-dE) > system.rand_pool(i, t)){
228 
229  //update dE (spatial direction)
230  system.dE.col(t) += 4 * spins(i, t) * (system.interaction.row(i).transpose().cwiseProduct(spins.col(t)));
231  system.dE(i, t) *= -1;
232 
233  //update spins
234  spins(i, t) *= -1;
235  }
236  }

239  inline static std::size_t mod_t(std::int64_t a, std::size_t num_trotter_slices){
240  //a -> [-1:num_trotter_slices]
241  //return a%num_trotter_slices (a>0), num_trotter_slices-1 (a==-1)
242  return (a+num_trotter_slices)%num_trotter_slices;
243  }

285  inline static void update_spin(ClPIsing &system,
286  RandomNumberEngine& random_number_engine,
287  const utility::ClassicalUpdaterParameter& parameter) {
288 
289  auto urd = std::uniform_real_distribution<>(0, 1.0);
290  for (const auto &index_spin: system.get_active_variables()) {
291  if (system.dE(index_spin) <= 0.0 || std::exp(-parameter.beta*system.dE(index_spin)) > urd(random_number_engine)) {
292  system.update_spin_system(index_spin);
293  }
294  }
