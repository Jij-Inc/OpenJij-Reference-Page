68  ContinuousTimeIsing(const SpinConfiguration& init_spin_config,
69  const GraphType& init_interaction,
70  const double gamma)
71  : spin_config(init_spin_config),
72  num_spins(init_spin_config.size()+1),
73  interaction(utility::gen_matrix_from_graph<Eigen::RowMajor>(init_interaction)),
74  gamma(gamma) {
75 
76  assert(init_spin_config.size() == init_interaction.get_num_spins());
77 
78  // insert numbers to diagnonal elements in the interaction
79  SparseMatrixXx diag(init_interaction.get_num_spins()+1, init_interaction.get_num_spins()+1);
80 
81  for(typename SparseMatrixXx::InnerIterator it(interaction, init_interaction.get_num_spins()); it; ++it) {
82  std::size_t j = it.index();
83  FloatType v = it.value();
84  if(j != init_interaction.get_num_spins()){
85  diag.insert(j,j) = v;
86  }
87  else{
88  diag.insert(j,j) = -1;
89  }
90  }
91 
92 
93  interaction += diag;
94 
95  spin_config.push_back(std::vector<CutPoint> { CutPoint(0.0, 1) });
96  // initialize auxiliary spin with 1 along entire timeline
97  }

174  size_t get_temporal_spin_index(graph::Index site_index, TimeType time_point) const {
175  static const auto first_lt = [](CutPoint x, CutPoint y) { return x.first < y.first; };
176  // function to compare two time points (lt; less than)
177 
178  const auto& timeline = this->spin_config[site_index];
179  const auto dummy_cut = CutPoint(time_point, 0); // dummy variable for binary search
180  auto found_itr = std::upper_bound(timeline.begin(),
181  timeline.end(),
182  dummy_cut,
183  first_lt);
184 
185  if(found_itr == timeline.begin()) { // if the time_point lies before any time points
186  found_itr = timeline.end() - 1; // periodic boundary condition
187  } else {
188  found_itr--;
189  }
190 
191  return std::distance(timeline.begin(), found_itr);
192  }
