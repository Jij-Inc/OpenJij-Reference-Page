114  inline void _checkpair(Index idx1, Index idx2) const{
115  std::int64_t r1,c1,r2,c2;
116  std::tie(r1,c1) = to_rc(idx1);
117  std::tie(r2,c2) = to_rc(idx2);
118  if(
119  !((r1 == r2 && std::abs(c1 - c2) == 1) || //horizontal connect
120  (c1 == c2 && std::abs(r1 - r2) == 1) || //vertical connect
121  (r1 == r2 && c1 == c2)) //local field
122  ){
123  throw std::runtime_error("invalid index pair "+std::to_string(idx1)+" "+std::to_string(idx2)+" inserted in Square");
124  }
125  }

138  Index to_ind(std::int64_t r, std::int64_t c) const{
139  if(!(-1 <= r && r <= (std::int64_t)_num_row)){
140  throw std::runtime_error("invalid value r="+std::to_string(r)+" inserted in Square");

144  throw std::runtime_error("invalid value c="+std::to_string(c)+" inserted in Square");

174  Square(std::size_t num_row, std::size_t num_column, FloatType init_val=0)
175  : Sparse<FloatType>(num_row*num_column, 4+1), _init_val(init_val), _num_row(num_row), _num_column(num_column){
176  assert(num_row >= 1);
177  assert(num_column >= 1);
178 
179  for(std::size_t r=0; r<_num_row; r++){
180  for(std::size_t c=0; c<_num_column; c++){
181  //open boundary
182  if(r > 0){
183  //MINUS_R
184  this->Sparse<FloatType>::J(to_ind(r,c), to_ind(r-1,c)) = _init_val;
185  }
186  if(c > 0){
187  //MINUS_C
188  this->Sparse<FloatType>::J(to_ind(r,c), to_ind(r,c-1)) = _init_val;
189  }
190  if(r < num_row-1){
191  //PLUS_R
192  this->Sparse<FloatType>::J(to_ind(r,c), to_ind(r+1,c)) = _init_val;
193  }
194  if(c < num_column-1){
195  //PLUS_C
196  this->Sparse<FloatType>::J(to_ind(r,c), to_ind(r,c+1)) = _init_val;
197  }
198  //local field
199  this->Sparse<FloatType>::J(to_ind(r,c), to_ind(r,c)) = _init_val;
200  }
201  }

212  Square(const json& j, std::size_t num_row, std::size_t num_column, FloatType init_val=0) : Square<FloatType>(num_row, num_column, init_val){
213  if(!(j["num_variables"] <= num_row*num_column)){
214  throw std::runtime_error("number of system size does not match");
215  }
216  //define bqm with ising variables
217  auto bqm = json_parse<FloatType, cimod::Sparse>(j, false);
218  //interactions
219  for(auto&& elem : bqm.get_quadratic()){
220  const auto& key = elem.first;
221  const auto& val = elem.second;
222  _checkpair(key.first, key.second);
223  this->Sparse<FloatType>::J(key.first, key.second) += val;
224  }
225  //local field
226  for(auto&& elem : bqm.get_linear()){
227  const auto& key = elem.first;
228  const auto& val = elem.second;
229  this->Sparse<FloatType>::h(key) += val;
230  }
