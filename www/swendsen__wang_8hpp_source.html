46  using ClIsing = system::ClassicalIsing<graph::Sparse<FloatType>>;
47 
48  template<typename RandomNumberEngine>
49  inline static void update(ClIsing& system,
50  RandomNumberEngine& random_number_engine,
51  const utility::ClassicalUpdaterParameter& parameter) {
52  auto urd = std::uniform_real_distribution<>(0, 1.0);
53 
54  // num_spin = system size + additional spin
55  const size_t num_spin = system.spin.size();
56 
57  // 1. update bonds
58  auto union_find_tree = utility::UnionFind(num_spin);
59  for (std::size_t node = 0; node < num_spin; ++node) {
60  for (typename ClIsing::SparseMatrixXx::InnerIterator it(system.interaction, node); it; ++it) {
61  //fetch adjacent node
62  std::size_t adj_node = it.index();
63  //fetch system.interaction(node, adj_node)
64  const FloatType& J = it.value();
65  if (node >= adj_node) continue;
66  //check if bond can be connected
67  if (J * system.spin(node) * system.spin(adj_node) > 0) continue;
68  const auto unite_rate = std::max(static_cast<FloatType>(0.0), static_cast<FloatType>(1.0 - std::exp( - 2.0 * parameter.beta * std::abs(J))));
69  if (urd(random_number_engine) < unite_rate)
70  union_find_tree.unite_sets(node, adj_node);
71  }
72  }
73 
74  // 2. make clusters
75  const auto cluster_map = [num_spin, &union_find_tree](){
76  auto cluster_map = std::unordered_multimap<utility::UnionFind::Node, utility::UnionFind::Node>();
77  for (std::size_t node = 0; node < num_spin; ++node) {
78  cluster_map.insert({union_find_tree.find_set(node), node});
79  }
80  return cluster_map;
81  }();
82 
83  // 3. update spin states in each cluster
84  for (auto&& c : union_find_tree.get_roots()) {
85  const auto range = cluster_map.equal_range(c);
86 
87  // 3.1. decide spin state (flip with the probability 1/2)
88  const FloatType probability = 1.0 / 2.0;
89  if(urd(random_number_engine) < probability){
90  // 3.2. update spin states
91  for (auto itr = range.first, last = range.second; itr != last; ++itr) {
92  const auto idx = itr->second;
93  system.spin(idx) *= -1;
94  }
95  }
96  }
97 
98  return;
99  }
100  };
