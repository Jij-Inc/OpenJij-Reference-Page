206  }),

208  .def("get_num_interactions", &Poly::get_num_interactions)
209  .def("calc_energy", &Poly::calc_energy, "spins"_a, "omp_flag"_a = true)
210  .def("energy", &Poly::energy, "spins"_a, "omp_flag"_a = true)
211  .def(
212  "__setitem__",
213  [](Poly &self, graph::Index key, FloatType val) {
214  self.J(key) = val;
215  },
216  "key"_a, "val"_a)
217  .def(
218  "__setitem__",
219  [](Poly &self, std::vector<graph::Index> &key, FloatType val) {
220  self.J(key) = val;
221  },
222  "key"_a, "val"_a)
223  .def(
224  "__getitem__",
225  [](const Poly &self, std::vector<graph::Index> &key) {
226  return self.J(key);
227  },
228  "key"_a)
229  .def(
230  "__getitem__",
231  [](const Poly &self, graph::Index key) { return self.J(key); },
232  "key"_a)
233  .def("get_polynomial", [](const Poly &self) {
234  py::dict py_polynomial;
235  for (std::size_t i = 0; i < self.get_keys().size(); ++i) {
236  py::tuple temp;
237  for (const auto &it : self.get_keys()[i]) {
238  temp = temp + py::make_tuple(it);
239  }
240  py_polynomial[temp] = self.get_values()[i];
241  }
242  return py_polynomial;
243  });
244 }

476 template <typename GraphType>
477 inline void declare_KLocalPolynomial(py::module &m,
478  const std::string &gtype_str) {
479 
480  using KLP = system::KLocalPolynomial<GraphType>;
481  auto str = std::string("KLocal") + gtype_str;
482 
483  py::class_<KLP>(m, str.c_str(), py::module_local())
484  .def(py::init<const graph::Binaries &, const GraphType &>(),
485  "init_spin"_a, "init_interaction"_a)
486  .def(py::init(
487  [](const graph::Binaries &init_binaries, const py::object &obj) {
488  return std::unique_ptr<KLP>(
489  new KLP(init_binaries, static_cast<nlohmann::json>(obj)));
490  }),
491  "init_binaries"_a, "obj"_a)
492  .def_readonly("binaries", &KLP::binaries)
493  .def_readonly("num_binaries", &KLP::num_binaries)
494  .def_readonly("count_call_updater", &KLP::count_call_updater)
495  .def_property_readonly("num_interactions", &KLP::GetNumInteractions)
496  .def_readwrite("rate_call_k_local", &KLP::rate_call_k_local)
497  .def("reset_binaries", &KLP::reset_binaries, "init_binaries"_a)
498  .def("reset_spins", &KLP::reset_binaries, "init_spins"_a)
499  .def("reset_dE", &KLP::reset_dE)
500  .def("get_active_binaries", &KLP::get_active_binaries)
501  .def("get_max_effective_dE", &KLP::get_max_effective_dE)
502  .def("get_min_effective_dE", &KLP::get_min_effective_dE)
503  .def("get_keys", &KLP::get_keys)
504  .def("get_values", &KLP::get_values)
505  .def("get_vartype_to_string", &KLP::get_vartype_string)
506  .def("get_polynomial",
507  [](const KLP &self) {
508  py::dict py_polynomial;
509  const auto &poly_key_list = self.get_keys();
510  const auto &poly_value_list = self.get_values();
511  for (std::size_t i = 0; i < poly_key_list.size(); ++i) {
512  py::tuple tuple;
513  for (const auto &index : poly_key_list[i]) {
514  tuple = tuple + py::make_tuple(index);
515  }
516  py_polynomial[tuple] = poly_value_list[i];
517  }
518  return py_polynomial;
519  })
520  .def("get_adj", [](const KLP &self) {
521  const auto &adj = self.get_adj();
522  const auto &poly_key_list = self.get_keys();
523  const auto &poly_value_list = self.get_values();
524  py::dict py_adj;
525  for (int64_t i = 0; i < self.num_binaries; ++i) {
526  py::dict dict;
527  for (const auto &index_key : adj[i]) {
528  py::tuple tuple;
529  for (const auto &index_binary : poly_key_list[index_key]) {
530  tuple = tuple + py::make_tuple(index_binary);
531  }
532  dict[tuple] = poly_value_list[index_key];
533  }
534  py_adj[py::int_(i)] = dict;
